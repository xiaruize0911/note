#换根dp #贪心

由限制 $h_i$ 大于点的度数，最终回到根的答案必然是经过每个节点的

根的答案可以 $\mathcal{O}(n)$ 的算出

考虑如何换根，分 $3$ 种情况(假设现在由 $rt \rightarrow x$)

1. 当前的 $rt$ 有多余的出边，那么用这个出边走到 $x$，$ans+1$
2. 当前 $rt$ 没有多余的出边，那么将 $x$ 向 $rt$ 的边取消，这样 $x$ 就多出一条边
	1. 若 $x$ 的所有儿子都满了，没法接收这条边， $ans-1$
	2. 若 $x$ 有不满的儿子，那么将 $x\rightarrow rt$ 的边指向这个儿子，$ans+1$

void pre(int x, int fa)
{
	debug(x, val[x]);
	for (auto v : g[x])
	{
		if (v == fa)
			continue;
		pre(v, x);
		int tmp = min(val[x], val[v]);
		val[x] -= tmp;
		val[v] -= tmp;
		cur += tmp * 2;
		if (val[v])
			st[x] = v;
	}
	debug(x, val[x]);
}

void dfs(int x, int fa)
{
	res[x] = cur;
	for (auto v : g[x])
	{
		if (v == fa)
			continue;
		if (val[x])
		{
			val[x]--;
			cur++;
			dfs(v, x);
			val[x]++;
			cur--;
		}
		else if (st[v])
		{
			val[st[v]]++;
			cur++;
			dfs(v, x);
			val[st[v]]--;
			cur--;
		}
		else
		{
			cur--;
			val[v]++;
			dfs(v, x);
			val[v]--;
			cur++;
		}
	}
}

		cur += 2;
	}
	pre(1, 0);
	debug(cur);
	dfs(1, 0);

}
