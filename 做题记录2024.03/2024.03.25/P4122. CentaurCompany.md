const int N = 35 + 10;

int n, a[N];
vector<int> g[N];
int dp[N][N][N][2];
int tmp[N][N][2];
int siz[N];
double res;

void dfs(int x, int fa)
{
    siz[x] = 1;
    dp[x][0][0][0] = dp[x][1][1][1] = 1;
    for (auto v : g[x])
    {
        if (v == fa)
            continue;
        dfs(v, x);
        mms(tmp, 0);
        rep(i, 0, siz[x])
        {
            rep(j, i, siz[x])
            {
                rep(a, 0, siz[v])
                {
                    rep(b, a, siz[v])
                    {
                        tmp[i + a][j + b][0] += dp[x][i][j][0] * (dp[v][a][b][1] + dp[v][a][b][0]);
                        tmp[i + a][j + b][1] += dp[x][i][j][1] * dp[v][a][b][0];
                        tmp[i + a - 1][j + b][1] += dp[x][i][j][1] * dp[v][a][b][1];
                    }
                }
            }
        }
        siz[x] += siz[v];
        memcpy(dp[x], tmp, sizeof(tmp));
    }
}
