int n, m, k;
int fa1[N], fa2[N];
vector<int> g[N], f[N];
double res = 0;
int cnt1 = 0, cnt2 = 0;

void dfs(int x, int dep, int fa, int val)
{
    if (dep == val)
    {
        cnt1++;
        return;
    }
    for (auto v : g[x])
    {
        if (v == fa)
            continue;
        dfs(v, dep + 1, x, val);
    }
}

void dfs2(int x, int dep, int fa, int val)
{
    if (dep == val)
    {
        cnt2++;
        return;
    }
    for (auto v : f[x])
    {
        if (v == fa)
            continue;
        dfs2(v, dep + 1, x, val);
    }
}
