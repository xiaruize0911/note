#Nim游戏 

建树，考虑公平组合游戏

对于一个点 $SG(x)$，枚举它及它的子树，计算所有的后继状态，具体来说每个后继状态都是在这个子树中删除一条链，然后将分开的每一个子树的 $SG$ 异或起来，这个值为这个子状态的 $SG$ 值

然后 $SG(x)=mex(SG(v))$

特判最上面一层

void calc(int x, int val)
{
    for (auto v : g[x])
        val ^= sg[v];
    st.insert(val);
    for (auto v : g[x])
        calc(v, val ^ sg[v]);
}

void dfs(int x)
{
    // cerr << "flag" << x << endl;
    for (auto v : g[x])
        dfs(v);
    if (!x)
        return;
    st.clear();
    calc(x, 0);
    // cerr << x << endl;
    // for (auto v : st)
    //     cerr << v << ' ';
    // cerr << endl;
    while (st.find(sg[x]) != st.end())
        sg[x]++;
    // cerr << x << ' ' << sg[x] << endl;
}

void solve()
